create database bd_barberia;
use bd_barberia
create table usuario(
id_user int auto_increment not null primary key,
email varchar(50) not null,
usuario varchar(50) not null,
password varchar(50) not null
);
---procedimiento insertar usuarios
DELIMITER //
CREATE PROCEDURE sp_insert_user(
    IN p_email VARCHAR(50),
    IN p_usuario VARCHAR(50),
    IN p_password VARCHAR(50)
)
BEGIN
    INSERT INTO usuario(email, usuario, password) VALUES(p_email, p_usuario, p_password);
END //
DELIMITER ;

create table tb_rostro(
id_rostro int auto_increment not null primary key,
rostro varchar(50) not null,
caracteristicas varchar(500) not null
)
---procedimiento insertar rostros
DELIMITER //
CREATE PROCEDURE sp_insert_rostro(
in p_rostro varchar(50),
in p_caracteristicas varchar(500) 
)
BEGIN
    INSERT INTO tb_rostro(rostro, caracteristicas) VALUES(p_rostro, p_caracteristicas);
END //
